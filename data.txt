import os
import random
import subprocess
from datetime import datetime, timedelta


def get_repo_path(prompt, default="."):
    while True:
        user_input = input(f"{prompt} (default current directory): ")

        if not user_input.strip():
            return default

        if os.path.isdir(user_input):
            return user_input

        print("Directory does not exist. Please enter a valid path.")


def get_filename(prompt, default="data.txt"):
    user_input = input(f"{prompt} (default {default}): ")

    if not user_input.strip():
        return default

    return user_input


def generate_natural_commit_dates(years=5):
    today = datetime.now()
    start_date = today - timedelta(days=365 * years)

    commit_dates = []
    current_date = start_date

    while current_date <= today:
        week_active = random.random() < 0.28

        if not week_active:
            current_date += timedelta(days=7)
            continue

        active_days = random.choices(
            population=[1, 2, 3],
            weights=[35, 45, 20],
            k=1
        )[0]

        start_offset = random.randint(0, 4)
        work_start_date = current_date + timedelta(days=start_offset)

        for day_index in range(active_days):
            work_date = work_start_date + timedelta(days=day_index)

            if work_date > today:
                continue

            commits_today = random.choices(
                population=[
                    random.randint(4, 8),
                    random.randint(9, 14),
                    random.randint(15, 22),
                    random.randint(23, 30)
                ],
                weights=[15, 25, 40, 20],
                k=1
            )[0]

            for _ in range(commits_today):
                random_seconds = random.randint(
                    10 * 3600,
                    23 * 3600
                )

                commit_date = work_date + timedelta(
                    seconds=random_seconds
                )

                commit_dates.append(commit_date)

        current_date += timedelta(days=7)

    random.shuffle(commit_dates)
    return commit_dates


def make_commit(date, repo_path, filename):
    filepath = os.path.join(repo_path, filename)

    with open(filepath, "a", encoding="utf-8") as f:
        f.write(f"Update at {date.isoformat()}\\n")

    subprocess.run(
        ["git", "add", filename],
        cwd=repo_path,
        check=True
    )

    env = os.environ.copy()

    date_str = date.strftime("%Y-%m-%dT%H:%M:%S")

    env["GIT_AUTHOR_DATE"] = date_str
    env["GIT_COMMITTER_DATE"] = date_str

    commit_messages = [
        "fix",
        "update",
        "improve ui",
        "refactor",
        "cleanup",
        "new feature",
        "responsive fix",
        "bug fix",
        "optimize",
        "code improvements",
        "small changes",
        "project update"
    ]

    subprocess.run(
        [
            "git",
            "commit",
            "-m",
            random.choice(commit_messages)
        ],
        cwd=repo_path,
        env=env,
        check=True
    )


def main():
    print("=" * 60)
    print("🌱 Natural GitHub Contribution Generator")
    print("=" * 60)

    repo_path = get_repo_path(
        "Enter the path to your local git repository",
        "."
    )

    filename = get_filename(
        "Enter the filename to modify for commits",
        "data.txt"
    )

    commit_dates = generate_natural_commit_dates(years=5)

    print(f"\\nRepo: {repo_path}")
    print(f"File: {filename}")
    print(f"Total commits to create: {len(commit_dates)}\\n")

    confirm = input("Do you want to continue? Type yes: ")

    if confirm.lower() != "yes":
        print("Cancelled.")
        return

    for i, commit_date in enumerate(commit_dates):
        print(
            f"[{i + 1}/{len(commit_dates)}] "
            f"{commit_date.strftime('%Y-%m-%d %H:%M:%S')}"
        )

        make_commit(
            commit_date,
            repo_path,
            filename
        )

    print("\\nPushing commits to remote repository...")

    subprocess.run(
        ["git", "push"],
        cwd=repo_path,
        check=True
    )

    print("\\n✅ Done! Check your GitHub contribution graph later.")


if __name__ == "__main__":
    main()
